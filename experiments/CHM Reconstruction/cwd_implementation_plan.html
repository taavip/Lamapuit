import laspy, numpy as np

def audit_ground_density(laz_path: str, tile_area_m2: float = 676) -> dict:
    """Compute ground return density for a LAZ tile.
    Tile area: 26m × 26m = 676 m²
    """
    las = laspy.read(laz_path)
    ground_mask = las.classification == 2  # ASPRS class 2 = ground
    n_ground = ground_mask.sum()
    density = n_ground / tile_area_m2

    # Height percentiles of surviving low returns (0.1–2.0m)
    z = las.z[~ground_mask]
    z_norm = z - las.z[ground_mask].mean() if ground_mask.any() else z
    low_returns = z_norm[(z_norm > 0.05) & (z_norm < 2.5)]

    return {
        "ground_density_pts_m2": round(density, 3),
        "total_returns": len(las.x),
        "ground_returns": int(n_ground),
        "cwd_candidate_returns": len(low_returns),  # 5cm–2.5m above ground
        "viable_for_ai": density > 0.5
    }

def validate_chm_tile(tif_path: str) -> dict:
    with rasterio.open(tif_path) as src:
        assert src.crs.to_epsg() == 3301, f"CRS is {src.crs}, expected EPSG:3301"

            "shape": chm.shape,
            "valid_fraction": round(valid_frac, 4),
            "nodata_fraction": round(1 - valid_frac, 4),
            "height_p50": float(np.median(chm[valid_mask])),
            "height_p95": float(np.percentile(chm[valid_mask], 95)),

Tile Strategy & Mask Simulation

Step 3.1 — Tile size and overlap strategy

Raw 26m tiles (130×130 px) are small. Use a larger contextual window at training time with 50% stride overlap to handle boundary logs and increase effective dataset size.

def extract_patches(chm: np.ndarray, mask: np.ndarray,
                    patch_size=256, stride=128) -> list:
    """Extract overlapping patches. Padding ensures boundary coverage."""
    H, W = chm.shape
    pad = patch_size // 2
    chm_p  = np.pad(chm,  pad, mode='reflect')
    mask_p = np.pad(mask, pad, mode='constant', constant_values=0)

    patches = []
    for y in range(0, H, stride):
        for x in range(0, W, stride):
            chm_patch  = chm_p [y:y+patch_size, x:x+patch_size]
            mask_patch = mask_p[y:y+patch_size, x:x+patch_size]
            if mask_patch.mean() > 0.05:  # skip fully empty patches
                patches.append({"chm": chm_patch, "valid_mask": mask_patch,
                                 "origin": (y, x)})
    return patches

Step 3.2 — Realistic ALS mask simulation SOTA

This is your primary methodological contribution in data preparation. Masks must statistically resemble real ALS gap patterns, not random dropout. Three gap types are combined: clustered canopy occlusion, flight-line stripe gaps, and fine random loss.

import cv2
from scipy.ndimage import binary_dilation

def simulate_als_mask(shape: tuple, target_nodata: float = 0.97,
                      seed: int = None) -> np.ndarray:
    """Generate a realistic ALS NoData mask.

    Returns binary mask: 1 = valid data, 0 = NoData.
    target_nodata: fraction of pixels to mark as NoData.
    Mirrors three physical sources of LiDAR data loss.
    """
    rng = np.random.default_rng(seed)
    H, W = shape
    mask = np.ones(shape, dtype=np.float32)

    # --- 1. Canopy occlusion: clustered elliptical holes ---
    n_clusters = rng.integers(80, 200)
    for _ in range(n_clusters):
        cy, cx = rng.integers(0, H), rng.integers(0, W)
        ry = rng.integers(3, 18)
        rx = rng.integers(3, 25)
        angle = rng.uniform(0, 180)
        ellipse_mask = np.zeros(shape, dtype=np.uint8)
        cv2.ellipse(ellipse_mask, (cx, cy), (rx, ry), angle, 0, 360, 1, -1)
        mask[ellipse_mask == 1] = 0

    # --- 2. Flight-line stripes (occasional linear gaps) ---
    n_stripes = rng.integers(0, 4)
    for _ in range(n_stripes):
        x_start = rng.integers(0, W)
        width   = rng.integers(1, 5)
        mask[:, x_start:x_start+width] = 0

    # --- 3. Fine random dropout (sensor noise) ---
    dropout_frac = rng.uniform(0.05, 0.20)
    mask[rng.random(shape) < dropout_frac] = 0

    # --- Calibrate to target sparsity by dilating existing holes ---
    current_nodata = 1.0 - mask.mean()
    while current_nodata < target_nodata - 0.01:
        mask = 1.0 - binary_dilation(1 - mask).astype(np.float32)
        current_nodata = 1.0 - mask.mean()

    return mask
Citation: Justify this approach with reference to ALS gap characterisation literature: Morsdorf et al. (2006) on canopy gap fraction, and Leiterer et al. (2015) on sub-canopy point density. Your specific gap statistics (cluster size distribution, stripe frequency) should be derived from your own data in a calibration step — measure gap morphology on real tiles and parameterise the simulator accordingly.

Step 4.1 — Partial Convolution layer Liu et al. 2018

The core building block. The mask is updated through each layer (valid regions grow as context is established), preventing NoData pixels from corrupting features.

import torch, torch.nn as nn, torch.nn.functional as F

class PartialConv2d(nn.Module):
    def __init__(self, in_ch, out_ch, kernel_size=3, stride=1, padding=1):
        super().__init__()
        self.conv = nn.Conv2d(in_ch, out_ch, kernel_size, stride, padding, bias=False)
        self.mask_conv = nn.Conv2d(1, 1, kernel_size, stride, padding, bias=False)
        nn.init.constant_(self.mask_conv.weight, 1.0)
        for p in self.mask_conv.parameters():
            p.requires_grad = False
        self.bn = nn.BatchNorm2d(out_ch)

    def forward(self, x, mask):
        # mask: 1 = valid, 0 = NoData. Must be same spatial size as x.
        with torch.no_grad():
            mask_sum = self.mask_conv(mask)
            kernel_sum = self.mask_conv.weight.sum()
            update_mask = (mask_sum > 0).float()

        # Apply mask to input before convolution
        x_masked = x * mask
        out = self.conv(x_masked)

        # Normalise by number of valid pixels in each kernel window
        out = torch.where(
            update_mask.bool(),
            out * (kernel_sum / (mask_sum + 1e-8)),
            torch.zeros_like(out)
        )
        out = self.bn(F.relu(out))
        # Update mask: any window with ≥1 valid pixel → valid after this layer
        return out, update_mask

Step 4.2 — CBAM Attention bottleneck Woo et al. 2018

Applied at the bottleneck only. Channel attention re-weights feature maps by importance; spatial attention focuses computation on informative spatial locations (i.e. the sparse valid pixels).

class CBAM(nn.Module):
    def __init__(self, channels, reduction=16, kernel_size=7):
        super().__init__()
        # Channel attention
        self.avg_pool = nn.AdaptiveAvgPool2d(1)
        self.max_pool = nn.AdaptiveMaxPool2d(1)
        self.fc = nn.Sequential(
            nn.Linear(channels, channels // reduction, bias=False),
            nn.ReLU(),
            nn.Linear(channels // reduction, channels, bias=False)
        )
        # Spatial attention
        self.spatial = nn.Conv2d(2, 1, kernel_size, padding=kernel_size//2, bias=False)

    def forward(self, x):
        B, C, H, W = x.shape
        # Channel
        avg = self.fc(self.avg_pool(x).view(B, C))
        mx  = self.fc(self.max_pool(x).view(B, C))
        ch_att = torch.sigmoid(avg + mx).view(B, C, 1, 1)
        x = x * ch_att
        # Spatial
        sp_in = torch.cat([x.mean(dim=1, keepdim=True),
                           x.max(dim=1, keepdim=True).values], dim=1)
        sp_att = torch.sigmoid(self.spatial(sp_in))
        return x * sp_att

Step 4.3 — MC Dropout for uncertainty quantification Gal & Ghahramani 2016

Dropout is kept active at inference time. Running N forward passes and computing variance across them gives a pixel-wise uncertainty map. High-uncertainty CWD predictions are flagged in the output.

def predict_with_uncertainty(model, x, mask, n_passes=30):
    """Run N stochastic forward passes with dropout active.
    Returns mean prediction and epistemic uncertainty per pixel."""
    model.train()  # keeps dropout ON
    preds = []
    with torch.no_grad():
        for _ in range(n_passes):
            recon, seg = model(x, mask)
            preds.append(seg.unsqueeze(0))

    preds = torch.cat(preds, dim=0)
    mean_pred   = preds.mean(dim=0)
    uncertainty = preds.var(dim=0)   # epistemic uncertainty
    return mean_pred, uncertainty
PHASE 05

Loss Function Design

Step 5.1 — Reconstruction loss (L_recon)

Three components applied only to originally-valid pixels in the target tile. The Sobel edge term is critical for log edge preservation.

class ReconstructionLoss(nn.Module):
    def __init__(self, lambda_perc=0.1, lambda_edge=0.5):
        super().__init__()
        self.lambda_perc = lambda_perc
        self.lambda_edge = lambda_edge

        # Sobel kernels for edge loss
        sobel_x = torch.tensor([[-1,0,1],[-2,0,2],[-1,0,1]], dtype=torch.float32)
        sobel_y = sobel_x.T
        self.register_buffer('sobel_x', sobel_x.view(1,1,3,3))
        self.register_buffer('sobel_y', sobel_y.view(1,1,3,3))

    def sobel_edges(self, x):
        Gx = F.conv2d(x, self.sobel_x, padding=1)
        Gy = F.conv2d(x, self.sobel_y, padding=1)
        return torch.sqrt(Gx**2 + Gy**2 + 1e-8)

    def forward(self, pred, target, valid_mask):
        # Only compute loss on originally valid pixels
        m = valid_mask.float()

        l1 = (torch.abs(pred - target) * m).sum() / (m.sum() + 1e-8)

        edge_pred   = self.sobel_edges(pred)   * m
        edge_target = self.sobel_edges(target) * m
        l_edge = F.l1_loss(edge_pred, edge_target)

        return l1 + self.lambda_edge * l_edge

Step 5.2 — Segmentation loss (L_seg)

Combined Dice + Focal loss handles the extreme class imbalance (CWD pixels ≪ ground pixels). Dice is insensitive to class ratio; Focal down-weights easy negatives.

class SegmentationLoss(nn.Module):
    def __init__(self, gamma=2.0, smooth=1.0):
        super().__init__()
        self.gamma = gamma
        self.smooth = smooth

    def dice_loss(self, pred, target):
        pred_f   = pred.flatten(1)
        target_f = target.flatten(1)
        intersection = (pred_f * target_f).sum(1)
        return 1 - (2*intersection + self.smooth) / \
               (pred_f.sum(1) + target_f.sum(1) + self.smooth)

    def focal_loss(self, pred, target):
        bce = F.binary_cross_entropy(pred, target, reduction='none')
        pt  = torch.exp(-bce)
        return ((1 - pt) ** self.gamma * bce).mean()

    def forward(self, pred, target):
        return self.dice_loss(pred, target).mean() + self.focal_loss(pred, target)

Step 5.3 — Joint loss with learned weighting

Use uncertainty-based multi-task loss weighting (Kendall et al. 2018) so the balance between reconstruction and segmentation is learned rather than manually tuned — more defensible academically.

class MultiTaskLoss(nn.Module):
    def __init__(self):
        super().__init__()
        # log(σ²) — learned task uncertainty weights
        self.log_var_recon = nn.Parameter(torch.zeros(1))
        self.log_var_seg   = nn.Parameter(torch.zeros(1))

    def forward(self, l_recon, l_seg):
        # Uncertainty-weighted multi-task loss (Kendall et al. 2018)
        w_recon = torch.exp(-self.log_var_recon)
        w_seg   = torch.exp(-self.log_var_seg)

        loss = (w_recon * l_recon + self.log_var_recon +
                w_seg   * l_seg   + self.log_var_seg)
        return loss.mean(), {
            "weight_recon": w_recon.item(),
            "weight_seg": w_seg.item()
        }
Logging the learned weights per epoch is a meaningful experimental result for your thesis. If the model shifts weight from reconstruction to segmentation over training, this indicates the encoder has learned the spatial structure and is ready to focus on detection.

Step 6.1 — Curriculum learning strategy

Start training on easier (less sparse) masks and gradually increase sparsity. This mirrors human learning and prevents the model from collapsing early when faced with 99% NoData from epoch 1.

  Epochs  1–10:  target_nodata = 0.70  (relatively easy, model learns CHM structure)
  Epochs 11–20:  target_nodata = 0.85  (moderate difficulty)
  Epochs 21–35:  target_nodata = 0.95  (near real-world conditions)
  Epochs 36–50:  target_nodata = 0.99  (full challenge, real-world match)

Step 6.2 — PyTorch Lightning training module

import pytorch_lightning as pl

class CWDModel(pl.LightningModule):
    def __init__(self, encoder_name="efficientnet-b2", dropout_rate=0.3):
        super().__init__()
        self.model      = MultiTaskPartialUNet(encoder_name, dropout_rate)
        self.recon_loss = ReconstructionLoss(lambda_edge=0.5)
        self.seg_loss   = SegmentationLoss(gamma=2.0)
        self.mt_loss    = MultiTaskLoss()
        self.save_hyperparameters()

    def training_step(self, batch, batch_idx):
        chm, valid_mask, cwd_label, sim_mask = batch
        x = torch.cat([chm * sim_mask, sim_mask], dim=1)  # 2-channel input

        chm_hat, cwd_prob = self.model(x, sim_mask)

        l_r = self.recon_loss(chm_hat, chm, valid_mask)
        l_s = self.seg_loss(cwd_prob, cwd_label)
        loss, weights = self.mt_loss(l_r, l_s)

        self.log_dict({"train/loss": loss, "train/l_recon": l_r,
                       "train/l_seg": l_s, **{f"w/{k}": v for k,v in weights.items()}})
        return loss

    def configure_optimizers(self):
        opt = torch.optim.AdamW(self.parameters(), lr=3e-4, weight_decay=1e-4)
        sched = torch.optim.lr_scheduler.CosineAnnealingLR(opt, T_max=50)
        return [opt], [sched]

Step 7.1 — Overlapping tile inference with confidence blending

def predict_tile_overlap(model, chm_tile, valid_mask_tile,
                         patch_size=256, stride=128, n_mc=30):
    """Inference with 50% overlap and Gaussian blending to eliminate
    boundary artefacts. Returns mean CWD probability and uncertainty."""
    H, W = chm_tile.shape
    accum_prob = np.zeros((H, W), dtype=np.float64)
    accum_unc  = np.zeros((H, W), dtype=np.float64)
    accum_w    = np.zeros((H, W), dtype=np.float64)

    # Gaussian weight window — upweights patch centre, downweights edges
    window = gaussian_window(patch_size)

    for y in range(0, H - patch_size + 1, stride):
        for x in range(0, W - patch_size + 1, stride):
            patch = chm_tile[y:y+patch_size, x:x+patch_size]
            pmask = valid_mask_tile[y:y+patch_size, x:x+patch_size]
            inp   = prepare_input(patch, pmask)

            mean_p, unc = predict_with_uncertainty(model, inp, n_passes=n_mc)
            accum_prob[y:y+patch_size, x:x+patch_size] += mean_p * window
            accum_unc [y:y+patch_size, x:x+patch_size] += unc   * window
            accum_w   [y:y+patch_size, x:x+patch_size] += window

    cwd_prob    = accum_prob / (accum_w + 1e-8)
    uncertainty = accum_unc  / (accum_w + 1e-8)
    return cwd_prob, uncertainty

Step 7.2 — Formalised CWD candidate filter

Post-processing applies three sequential filters to keep only high-confidence, physically plausible detections.

from skimage import measure, morphology

def extract_cwd_candidates(cwd_prob, uncertainty, valid_mask_original,
                           prob_thresh=0.5, unc_thresh=0.15,
                           min_valid_fraction=0.15,
                           min_elongation=3.0) -> gpd.GeoDataFrame:
    """Extract CWD candidates with confidence filtering.

    Formalised criteria (state these explicitly in thesis methods):
      1. Probability threshold:     p_cwd > prob_thresh
      2. Uncertainty threshold:     uncertainty < unc_thresh
      3. Valid pixel fraction:      ≥15% of segment footprint has original ALS returns
      4. Shape elongation:          major_axis / minor_axis > 3.0
    """
    # 1+2: probability and uncertainty gates
    binary = (cwd_prob > prob_thresh) & (uncertainty < unc_thresh)

    # 3+4: per-segment shape and provenance filters
    labels = measure.label(binary)
    candidates = []

    for region in measure.regionprops(labels):
        # Shape filter: elongation
        if region.minor_axis_length < 1: continue
        elongation = region.major_axis_length / region.minor_axis_length
        if elongation < min_elongation: continue

        # Provenance filter: original valid pixel fraction
        seg_mask = labels == region.label
        valid_frac = (valid_mask_original[seg_mask]).mean()
        if valid_frac < min_valid_fraction: continue

        candidates.append({
            "geometry": region.bbox,
            "length_px": region.major_axis_length,
            "elongation": elongation,
            "mean_prob": cwd_prob[seg_mask].mean(),
            "mean_uncertainty": uncertainty[seg_mask].mean(),
            "valid_return_fraction": valid_frac,
            # Confidence class for reporting
            "confidence": "HIGH" if valid_frac > 0.30 else "MEDIUM"
        })
    return candidates

from skimage.metrics import structural_similarity, peak_signal_noise_ratio

def evaluate_reconstruction(pred, target, valid_mask):
    m = valid_mask.astype(bool)
    rmse = np.sqrt(((pred[m] - target[m])**2).mean())
    ssim = structural_similarity(pred, target, data_range=target.max())
    return {"RMSE": rmse, "SSIM": ssim}

def evaluate_segmentation(pred_prob, gt_mask, threshold=0.5):
    pred_bin = (pred_prob > threshold).astype(int)
    tp = (pred_bin * gt_mask).sum()
    fp = (pred_bin * (1 - gt_mask)).sum()
    fn = ((1 - pred_bin) * gt_mask).sum()
    precision = tp / (tp + fp + 1e-8)
    recall    = tp / (tp + fn + 1e-8)
    f1 = 2 * precision * recall / (precision + recall + 1e-8)
    iou = tp / (tp + fp + fn + 1e-8)
    return {"F1": f1, "Precision": precision,
            "Recall": recall, "IoU": iou}
